Components

var homeComponent = {
  controller: function(args) {
    // this.heading = 'Home';
    return {
      heading: 'Home'
    };
  },
  view: function(ctrl, args) {
    return m('div.home', [
      m('h1', ctrl.heading)
    ]);
  }
};

Routing

// Routing attempts to match in order, but
// I've found it to be pretty smart.
m.route(document.querySelector('#app'), '/', {
  '/': home,
  '/:param?anotherParam': home
});

AJAX & Promises

var promise = m.request({
  method: 'GET',
  url: '/api/stuff',
  initialValue: []
});
var anotherPromise = m.deferred().promise;
var everything = m.sync([promise, anotherPromise]);

everything.then(...)

Special Mention:

m.prop

function User(data) {
  this.name = m.prop(data.name);
}

var user = new User({name: 'Tim'});
JSON.stringify(user); // '{"name": "Tim"}'

var users = m.prop([]);
m.request({
  method: 'GET',
  url: 'api/users'
}).then(users);
