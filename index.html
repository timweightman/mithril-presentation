Components

// home.js
var m = require('mithril');

module.exports = {
  controller: function(args) {
    return {
      heading: 'Home'
    };
  },
  view: function(ctrl, args) {
    return m('div.home', [
      m('h1', ctrl.heading)
    ]);
  }
};

Routing

// app.js
var m = require('mithril');
var home = require('components/home');
var todos = require('components/todos');
var todoItem = require('components/todoItem');

// Routing does attempt to match in order
m.route(document.querySelector('#app'), '/', {
  '/': home,
  '/todos': todos,
  '/todos/:status': todoItem
});

AJAX & Promises

var m = require('mithril');

var todosPromise = m.request({

Special Mention:

m.prop

var m = require('mithril');

function User(data) {
  this.name = m.prop(data.name);
}

var user = new User({name: 'Tim'});
JSON.stringify(user); // '{"name": "Tim"}'

var userPromise = m.request({
  method: 'GET',
  url: 'api/users',
  type: User
}).then(function(users) {
  console.log(users()); // [{name: 'Tim'}, {name:'Bob'}]
});
